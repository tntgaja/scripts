--// Auto Farm Bandit (Advanced TP Fix)
local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local VirtualUser = game:GetService("VirtualUser")
local LocalPlayer = Players.LocalPlayer

-- UI Toggle
local ScreenGui = Instance.new("ScreenGui", LocalPlayer.PlayerGui)
local ToggleButton = Instance.new("TextButton", ScreenGui)
ToggleButton.Size = UDim2.new(0, 120, 0, 40)
ToggleButton.Position = UDim2.new(0, 10, 0, 200)
ToggleButton.Text = "Auto Farm: OFF"

local autoFarm = false
ToggleButton.MouseButton1Click:Connect(function()
    autoFarm = not autoFarm
    ToggleButton.Text = autoFarm and "Auto Farm: ON" or "Auto Farm: OFF"
end)

-- Find closest mob
local function getNearestBandit()
    local nearest, dist = nil, math.huge
    for _, mob in pairs(workspace.Enemies:GetChildren()) do
        if mob:FindFirstChild("HumanoidRootPart") and mob:FindFirstChild("Humanoid") and mob.Humanoid.Health > 0 then
            local d = (mob.HumanoidRootPart.Position - LocalPlayer.Character.HumanoidRootPart.Position).Magnitude
            if d < dist then
                dist = d
                nearest = mob
            end
        end
    end
    return nearest
end

-- Safe Teleport
local function safeTP(targetPos)
    LocalPlayer.Character.HumanoidRootPart.CFrame = CFrame.new(targetPos + Vector3.new(0, 6, 0))
end

-- Farming loop
RunService.Heartbeat:Connect(function()
    if autoFarm and LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
        local bandit = getNearestBandit()
        if bandit then
            local myHRP = LocalPlayer.Character.HumanoidRootPart
            local mobHRP = bandit.HumanoidRootPart
            local dist = (mobHRP.Position - myHRP.Position).Magnitude

            if dist > 50 then
                -- Nếu xa hơn 50 studs thì dịch chuyển tới
                safeTP(mobHRP.Position)
            else
                -- Nếu gần thì bay lơ lửng trên đầu
                myHRP.CFrame = mobHRP.CFrame * CFrame.new(0, 6, 0)
                VirtualUser:ClickButton1(Vector2.new())
            end
        end
    end
end)
