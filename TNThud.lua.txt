-- TNThud Pro - Redz Style UI
-- Clean Professional Layout

local Players = game:GetService("Players")
local TweenService = game:GetService("TweenService")
local RunService = game:GetService("RunService")

local player = Players.LocalPlayer

-- Remove old GUI
if player.PlayerGui:FindFirstChild("TNThudPro") then
    player.PlayerGui.TNThudPro:Destroy()
end

-- GUI
local gui = Instance.new("ScreenGui")
gui.Name = "TNThudPro"
gui.ResetOnSpawn = false
gui.Parent = player:WaitForChild("PlayerGui")

-- Main Frame
local main = Instance.new("Frame")
main.Size = UDim2.new(0,500,0,350)
main.Position = UDim2.new(0.5,-250,0.5,-175)
main.BackgroundColor3 = Color3.fromRGB(18,18,18)
main.Active = true
main.Draggable = true
main.Parent = gui
Instance.new("UICorner", main).CornerRadius = UDim.new(0,18)

-- RGB Top Bar
local topBar = Instance.new("Frame")
topBar.Size = UDim2.new(1,0,0,5)
topBar.BackgroundColor3 = Color3.fromRGB(255,0,0)
topBar.Parent = main
Instance.new("UICorner", topBar).CornerRadius = UDim.new(0,18)

local hue = 0
RunService.RenderStepped:Connect(function()
    hue += 0.002
    if hue > 1 then hue = 0 end
    topBar.BackgroundColor3 = Color3.fromHSV(hue,1,1)
end)

-- Title
local title = Instance.new("TextLabel")
title.Size = UDim2.new(1,0,0,50)
title.Position = UDim2.new(0,0,0,10)
title.BackgroundTransparency = 1
title.Text = "TNThud Pro Hub"
title.TextColor3 = Color3.fromRGB(255,255,255)
title.TextScaled = true
title.Parent = main

-- Sidebar
local sidebar = Instance.new("Frame")
sidebar.Size = UDim2.new(0,150,1,-60)
sidebar.Position = UDim2.new(0,0,0,60)
sidebar.BackgroundColor3 = Color3.fromRGB(22,22,22)
sidebar.Parent = main
Instance.new("UICorner", sidebar).CornerRadius = UDim.new(0,15)

-- Content Area
local content = Instance.new("Frame")
content.Size = UDim2.new(1,-170,1,-70)
content.Position = UDim2.new(0,160,0,60)
content.BackgroundTransparency = 1
content.Parent = main

-- Tab System
local currentTab = nil

local function createTab(name,order)
    local btn = Instance.new("TextButton")
    btn.Size = UDim2.new(1,-20,0,45)
    btn.Position = UDim2.new(0,10,0,(order-1)*55+10)
    btn.BackgroundColor3 = Color3.fromRGB(30,30,30)
    btn.Text = name
    btn.TextColor3 = Color3.fromRGB(255,255,255)
    btn.TextScaled = true
    btn.Parent = sidebar
    Instance.new("UICorner", btn).CornerRadius = UDim.new(0,10)

    local page = Instance.new("Frame")
    page.Size = UDim2.new(1,0,1,0)
    page.BackgroundTransparency = 1
    page.Visible = false
    page.Parent = content

    btn.MouseButton1Click:Connect(function()
        if currentTab then
            currentTab.Visible = false
        end
        page.Visible = true
        currentTab = page

        TweenService:Create(btn,TweenInfo.new(0.2),{
            BackgroundColor3 = Color3.fromRGB(45,45,45)
        }):Play()
    end)

    return page
end

-- Create Tabs
local mainTab = createTab("Main",1)
local settingsTab = createTab("Settings",2)
local infoTab = createTab("Info",3)

mainTab.Visible = true
currentTab = mainTab

-- Professional Button
local function createButton(parent,text,posY)
    local btn = Instance.new("TextButton")
    btn.Size = UDim2.new(0.9,0,0,50)
    btn.Position = UDim2.new(0.05,0,0,posY)
    btn.BackgroundColor3 = Color3.fromRGB(28,28,28)
    btn.Text = text
    btn.TextColor3 = Color3.fromRGB(255,255,255)
    btn.TextScaled = true
    btn.Parent = parent
    Instance.new("UICorner", btn).CornerRadius = UDim.new(0,12)

    btn.MouseEnter:Connect(function()
        TweenService:Create(btn,TweenInfo.new(0.15),{
            BackgroundColor3 = Color3.fromRGB(40,40,40)
        }):Play()
    end)

    btn.MouseLeave:Connect(function()
        TweenService:Create(btn,TweenInfo.new(0.15),{
            BackgroundColor3 = Color3.fromRGB(28,28,28)
        }):Play()
    end)
end

-- Add sample content
createButton(mainTab,"Start Function",10)
createButton(mainTab,"Enable Feature",70)
createButton(settingsTab,"Reset Settings",10)

local infoLabel = Instance.new("TextLabel")
infoLabel.Size = UDim2.new(1,0,0,100)
infoLabel.BackgroundTransparency = 1
infoLabel.Text = "TNThud Pro UI\nProfessional Redz Style Layout"
infoLabel.TextColor3 = Color3.fromRGB(200,200,200)
infoLabel.TextScaled = true
infoLabel.Parent = infoTab
