-- TNThud MAX - Ultra Large Professional Hub

local Players = game:GetService("Players")
local TweenService = game:GetService("TweenService")
local RunService = game:GetService("RunService")

local player = Players.LocalPlayer

-- Remove old
if player.PlayerGui:FindFirstChild("TNThudMAX") then
    player.PlayerGui.TNThudMAX:Destroy()
end

-- GUI
local gui = Instance.new("ScreenGui")
gui.Name = "TNThudMAX"
gui.ResetOnSpawn = false
gui.Parent = player:WaitForChild("PlayerGui")

-- MAIN (BIGGER)
local main = Instance.new("Frame")
main.Size = UDim2.new(0,700,0,480)
main.Position = UDim2.new(0.5,-350,0.5,-240)
main.BackgroundColor3 = Color3.fromRGB(15,15,15)
main.Active = true
main.Draggable = true
main.Parent = gui
Instance.new("UICorner", main).CornerRadius = UDim.new(0,22)

-- RGB Border
local stroke = Instance.new("UIStroke")
stroke.Thickness = 4
stroke.Parent = main

-- RGB Top Bar
local topBar = Instance.new("Frame")
topBar.Size = UDim2.new(1,0,0,6)
topBar.Parent = main
Instance.new("UICorner", topBar).CornerRadius = UDim.new(0,22)

local hue = 0
RunService.RenderStepped:Connect(function()
    hue += 0.002
    if hue > 1 then hue = 0 end
    local color = Color3.fromHSV(hue,1,1)
    stroke.Color = color
    topBar.BackgroundColor3 = color
end)

-- TITLE
local title = Instance.new("TextLabel")
title.Size = UDim2.new(1,0,0,60)
title.Position = UDim2.new(0,0,0,10)
title.BackgroundTransparency = 1
title.Text = "TNThud MAX HUB"
title.TextColor3 = Color3.fromRGB(255,255,255)
title.TextScaled = true
title.Parent = main

-- SIDEBAR (WIDER)
local sidebar = Instance.new("Frame")
sidebar.Size = UDim2.new(0,200,1,-80)
sidebar.Position = UDim2.new(0,10,0,70)
sidebar.BackgroundColor3 = Color3.fromRGB(20,20,20)
sidebar.Parent = main
Instance.new("UICorner", sidebar).CornerRadius = UDim.new(0,18)

-- CONTENT
local content = Instance.new("Frame")
content.Size = UDim2.new(1,-230,1,-90)
content.Position = UDim2.new(0,220,0,75)
content.BackgroundTransparency = 1
content.Parent = main

local currentTab = nil

-- TAB FUNCTION
local function createTab(name,order)
    local btn = Instance.new("TextButton")
    btn.Size = UDim2.new(1,-20,0,55)
    btn.Position = UDim2.new(0,10,0,(order-1)*65+15)
    btn.BackgroundColor3 = Color3.fromRGB(30,30,30)
    btn.Text = name
    btn.TextColor3 = Color3.fromRGB(255,255,255)
    btn.TextScaled = true
    btn.Parent = sidebar
    Instance.new("UICorner", btn).CornerRadius = UDim.new(0,14)

    local page = Instance.new("Frame")
    page.Size = UDim2.new(1,0,1,0)
    page.BackgroundTransparency = 1
    page.Visible = false
    page.Parent = content

    btn.MouseButton1Click:Connect(function()
        if currentTab then
            currentTab.Visible = false
        end
        page.Visible = true
        currentTab = page
        
        TweenService:Create(btn,TweenInfo.new(0.2),{
            BackgroundColor3 = Color3.fromRGB(45,45,45)
        }):Play()
    end)

    return page
end

-- CREATE TABS
local mainTab = createTab("Main",1)
local settingsTab = createTab("Settings",2)
local visualTab = createTab("Visual",3)
local infoTab = createTab("Info",4)

mainTab.Visible = true
currentTab = mainTab

-- PROFESSIONAL BUTTON
local function createButton(parent,text,posY)
    local btn = Instance.new("TextButton")
    btn.Size = UDim2.new(0.9,0,0,65)
    btn.Position = UDim2.new(0.05,0,0,posY)
    btn.BackgroundColor3 = Color3.fromRGB(25,25,25)
    btn.Text = text
    btn.TextColor3 = Color3.fromRGB(255,255,255)
    btn.TextScaled = true
    btn.Parent = parent
    Instance.new("UICorner", btn).CornerRadius = UDim.new(0,16)

    btn.MouseEnter:Connect(function()
        TweenService:Create(btn,TweenInfo.new(0.15),{
            BackgroundColor3 = Color3.fromRGB(40,40,40)
        }):Play()
    end)

    btn.MouseLeave:Connect(function()
        TweenService:Create(btn,TweenInfo.new(0.15),{
            BackgroundColor3 = Color3.fromRGB(25,25,25)
        }):Play()
    end)
end

-- SAMPLE CONTENT
createButton(mainTab,"Start Feature",20)
