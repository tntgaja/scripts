-- Tab chính
local Tab1o = MakeTab({Name = "Auto Farm"})

-- Toggle Auto Farm Bandit
AddToggle(Tab1o, {
    Name = "Auto Farm Bandit (Nâng Cao)",
    Default = false,
    Callback = function(Value)
        _G.AutoFarmBandit = Value
        while _G.AutoFarmBandit do task.wait()
            AutoQuest()
            AutoEquip()
            AutoFarmBandit()
        end
    end
})

-- Hàm cầm vũ khí
function AutoEquip()
    local Weapon = {"Katana","Combat"}
    for _,v in pairs(Weapon) do
        if game.Players.LocalPlayer.Backpack:FindFirstChild(v) then
            game.Players.LocalPlayer.Character.Humanoid:EquipTool(
                game.Players.LocalPlayer.Backpack[v]
            )
        end
    end
end

-- Hàm nhận quest
function AutoQuest()
    local Quest = game.Players.LocalPlayer.PlayerGui.Main.Quest
    if not Quest.Visible then
        game.Players.LocalPlayer.Character.HumanoidRootPart.CFrame =
            CFrame.new(1060,16,1548)
        wait(1)
        -- Tương tác NPC Quest
    end
end

-- Hàm farm Bandit
function AutoFarmBandit()
    local Bandit = game.Workspace.Enemies:FindFirstChild("Bandit")
    if Bandit and Bandit:FindFirstChild("HumanoidRootPart") then
        local Root = game.Players.LocalPlayer.Character.HumanoidRootPart
        Root.CFrame = Bandit.HumanoidRootPart.CFrame * CFrame.new(0,10,0)
        -- đánh quái
        game:GetService("VirtualUser"):Button1Down(Vector2.new(0,0))
        wait(0.1)
        game:GetService("VirtualUser"):Button1Up(Vector2.new(0,0))
    else
        -- Không có quái thì quay về spawn Bandit
        game.Players.LocalPlayer.Character.HumanoidRootPart.CFrame =
            CFrame.new(1150,16,1600)
    end
end
