-- Auto Farm Bandit (Dùng Remote Attack)
local autofarm = false
local plr = game.Players.LocalPlayer
local rs = game:GetService("ReplicatedStorage")

-- Hàm nhận nhiệm vụ Bandit
function GetQuest()
    local args = {
        [1] = "StartQuest",
        [2] = "BanditQuest1",
        [3] = 1
    }
    rs.Remotes.CommF_:InvokeServer(unpack(args))
end

-- Hàm đánh quái bằng Skill/Attack remote
function Attack(enemy)
    -- gọi combat remote của game (đòn đánh thường)
    rs.Remotes.Combat:FireServer(enemy.HumanoidRootPart)
end

function AutoFarm()
    -- Kiểm tra có quest chưa, nếu chưa thì nhận
    local quest = plr.PlayerGui:FindFirstChild("Main") and plr.PlayerGui.Main:FindFirstChild("Quest")
    if not quest or not quest.Visible then
        GetQuest()
        task.wait(1)
    end

    -- Tìm và đánh Bandit
    for _,enemy in pairs(workspace.Enemies:GetChildren()) do
        if enemy.Name:match("Bandit") and enemy:FindFirstChild("HumanoidRootPart") and enemy:FindFirstChild("Humanoid") and enemy.Humanoid.Health > 0 then
            plr.Character.HumanoidRootPart.CFrame = enemy.HumanoidRootPart.CFrame * CFrame.new(0,0,3)
            -- spam tấn công
            for i=1,5 do
                Attack(enemy)
                task.wait(0.2)
            end
        end
    end
end

-- Nút bật/tắt
AddToggle(Tab1o, {
    Name = "Auto Farm Bandit",
    Default = false,
    Callback = function(Value)
        autofarm = Value
        if autofarm then
            spawn(function()
                while autofarm do
                    AutoFarm()
                    task.wait(0.5)
                end
            end)
        end
    end
})
