-- local autofarm = false
local VirtualUser = game:GetService("VirtualUser")

-- Hàm tìm Bandit còn sống
local function GetBandit()
    for _, mob in pairs(workspace.Enemies:GetChildren()) do
        if mob.Name == "Bandit" and mob:FindFirstChild("Humanoid") and mob.Humanoid.Health > 0 then
            return mob
        end
    end
    return nil
end

-- Hàm đánh Bandit
local function AttackBandit(bandit)
    while bandit and bandit:FindFirstChild("Humanoid") and bandit.Humanoid.Health > 0 and autofarm do
        game.Players.LocalPlayer.Character.HumanoidRootPart.CFrame = bandit.HumanoidRootPart.CFrame + Vector3.new(0,0,2)
        VirtualUser:Button1Down(Vector2.new(0,0), workspace.CurrentCamera.CFrame)
        task.wait(0.1)
        VirtualUser:Button1Up(Vector2.new(0,0), workspace.CurrentCamera.CFrame)
        task.wait(0.3)
    end
end

-- Toggle Auto Farm Bandit
AddToggle(Tab1o, {
    Name = "Auto Farm Bandit",
    Default = false,
    Callback = function(Value)
        autofarm = Value
        if autofarm then
            spawn(function()
                while autofarm do
                    local bandit = GetBandit()
                    if bandit then
                        AttackBandit(bandit)
                    end
                    task.wait(0.5)
                end
            end)
        end
    end
})
